            render() {

                const q = quizData.questions[this.currentIdx];

                container.innerHTML = '';

                this.answered = false;

                        <span class="flex-1">${opt.text}</span>

                    `;

                    btn.onclick = () => this.handleAnswer(opt, btn);

                    container.appendChild(btn);

                });

            },


            handleAnswer(opt, btn) {

                if (this.answered) return;

                    this.score++;

                    btn.classList.add('bg-emerald-50', 'border-emerald-500', 'ring-2', 'ring-emerald-100');

                } else {

                    btn.classList.add('bg-rose-50', 'border-rose-500', 'ring-2', 'ring-rose-100');

                    btn.querySelector('span').classList.add('bg-rose-500', 'text-white', 'border-rose-500');

                }


                const feedbackContainer = document.getElementById('feedback-container');

                const feedbackText = document.getElementById('feedback-text');

                const nextBtn = document.getElementById('next-btn');

                

                nextBtn.innerText = this.currentIdx === quizData.questions.length - 1 ? "See My Score" : "Next Question";

                feedbackContainer.classList.remove('hidden');

                feedbackContainer.classList.add(opt.isCorrect ? 'bg-emerald-50' : 'bg-slate-50');

                feedbackContainer.classList.add(opt.isCorrect ? 'border-emerald-100' : 'border-slate-200');

                feedbackText.innerHTML = `<strong>${opt.isCorrect ? 'Correct!' : 'Not quite.'}</strong> ${opt.rationale}`;

            },

                const q = quizData.questions[this.currentIdx];

                const feedbackContainer = document.getElementById('feedback-container');

                const feedbackText = document.getElementById('feedback-text');

                feedbackContainer.classList.remove('hidden');

                feedbackContainer.classList.remove('bg-emerald-50', 'border-emerald-100');

                feedbackContainer.classList.add('bg-blue-50', 'border-blue-100');

                feedbackText.innerHTML = `<strong>Hint:</strong> ${q.hint}`;

            },

                this.currentIdx++;

                    this.render();

                } else {

                    this.finish();

                }

                document.getElementById('quiz-screen').classList.add('hidden');
